定义一个函数，输入一个链表的头节点，反转该链表并输出反转后链表的头节点。
示例:
```
输入: 1->2->3->4->5->NULL
输出: 5->4->3->2->1->NULL
```
限制：
`0 <= 节点个数 <= 5000`

```js
// 迭代
// 时间复杂度O(n)
// 空间复杂度O(1)
var reverseList = function(head) {
    let prev  = null
    let curr = head
    while(curr) {
        const next = curr.next
        curr.next = prev
        prev = curr
        curr = next
    }
    return prev
};

// 递归
// 时间复杂度O(n)
// 空间复杂度O(n)
var reverseList = function (head) {
    if (head === null || head.next === null) return head;
    const newHead = reverseList(head.next);
    head.next.next = head;
    head.next = null;
    return newHead;
};
```
